def euler(func, initial, deltas):
    vals = [initial]
    for i in range(1, len(deltas)):
        cur = np.array(vals[-1][:])
        dt = deltas[i] - deltas[i - 1]
        cur += func(cur, dt)
        vals.append(cur)
    return np.array(vals)

    x, y, z = u

    dy = (-g * y  + d * x * y) * dt
    dz = (-n * z + m * z * x) * dt
    return [dx, dy, dz]
